% Simple bibliography
\clearpage
\phantomsection
\chapter*{\ReferencesHead}
\chaptermark{\ReferencesHead}
\printbibliography[heading=none]
\nocite{*}
\addcontentsline{toc}{chapter}{\ReferencesHead}
